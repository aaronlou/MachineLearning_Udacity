from sklearn.model_selection import KFold
from sklearn.tree import DecisionTreeRegressor
from sklearn.model_selection import GridSearchCV
from sklearn.metrics import make_scorer 


def fit_model(X, y):
    """  [X,y]"""
    
    cross_validator = KFold(n_splits=3,shuffle=True,random_state=30).split(X)
    
    regressor = DecisionTreeRegressor()

    params = {'max_depth': list(xrange(1,11)) }

    scoring_func = make_scorer(performance_metric)

    grid = GridSearchCV(regressor,params,scoring_func,cv=cross_validator)

    #  [X,y]
    grid = grid.fit(X, y)
    # print pd.DataFrame(grid.cv_results_)
    # 
    return grid.best_estimator_
